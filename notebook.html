def fun1(x):
    if x>1:
        return 1
    else:
        return 0


def fun2(A,x):
    'A is dataframe'
    'x=list of columns for which count is needed'
    f2=A
    for i in x:
        a=pd.DataFrame(A[i].value_counts()).reset_index()
        a.columns=[i,i+'_count']
        f2=pd.merge(f2,a,on=i,how='left')
        f2.drop(i,axis=1,inplace=True)
    return f2
